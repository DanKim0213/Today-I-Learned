## Solution 

Linked list which is implemented by dict() helps you to solve this problem. 

```python

def solution(n, k, cmd):
    cur = k
    table = {i: [i - 1, i + 1] for i in range(n)}
    answer = ["O"] * n
    table[0] = [None, 1]
    table[n - 1] = [n - 2, None]
    stack = []
    for c in cmd:
        if c == "C":
            # 삭제
            answer[cur] = "X"
            prev, next = table[cur]
            stack.append([prev, cur, next])
            if next is None:
                cur = table[cur][0]
            else:
                cur = table[cur][1]
            if prev is None:
                table[next][0] = None
            elif next is None:
                table[prev][1] = None
            else:
                table[prev][1] = next
                table[next][0] = prev

        elif c == "Z":
            # 복구
            prev, now, next = stack.pop()
            answer[now] = "O"
            if prev is None:
                table[next][0] = now
            elif next is None:
                table[prev][1] = now
            else:
                table[next][0] = now
                table[prev][1] = now

        else:
            # 커서 이동
            c1, c2 = c.split(" ")
            c2 = int(c2)
            if c1 == "D":
                for _ in range(c2):
                    cur = table[cur][1]
            else:
                for _ in range(c2):
                    cur = table[cur][0]
    return "".join(answer)
```

from collections import deque


def union(parents, x, y):
    parent_x = find_parent(parents, x)
    parent_y = find_parent(parents, y)

    if parent_x != parent_y:
        parents[y] = parent_x


def find_parent(parents, x):
    # out of range
    # FIXME:

    if parents[x] != x:
        parents[x] = find_parent(parents, parents[x])
    return parents[x]


def solution(n, k, cmd):
    parents = [i for i in range(n)]
    trash_bin = []  # stack
    queue = deque(cmd)

    while queue:
        command = queue.popleft().split()

        if command[0] == "U":
            for _ in range(command[1]):
                k = find_parent(parents, k - 1)
        elif command[0] == "D":
            # FIXME:
            for _ in range(command[1]):
                k = find_parent(parents, k + 1)
        elif command[0] == "C":
            # FIXME:
            new_k = find_parent(parents, k + 1)
            deprecated_k = find_parent(parents, k)
            union(parents, new_k, deprecated_k)
            k = new_k
            trash_bin.append(deprecated_k)
        else:  # Z
            deprecated_k = trash_bin.pop()
            parents[deprecated_k] = deprecated_k

    answer = "".join(["O" if parents[i] == i else "X" for i in parents])
    return answer
