from collections import deque

def build_time_dp(n, k, build_time, prerequisites, target):
    dp = [0] * (n+1)
    indegree = [0] * (n+1)
    graph = [[] for _ in range(n+1)]

    for x, y in prerequisites:
        graph[x].append(y)
        indegree[y] += 1

    queue = deque()

    for i in range(1, n+1):
        if indegree[i] == 0:
            queue.append(i)
            dp[i] = build_time[i]

    while queue:
        node = queue.popleft()

        if node == target:
            return dp[node]

        for adj_node in graph[node]:
            indegree[adj_node] -= 1
            dp[adj_node] = max(dp[adj_node], dp[node] + build_time[adj_node])

            if indegree[adj_node] == 0:
                queue.append(adj_node)

    return dp[target]


def main():
    t = int(input())

    for _ in range(t):
        n, k = map(int, input().split())
        build_time = [0] + list(map(int, input().split()))
        prerequisites = []

        for _ in range(k):
            x, y = map(int, input().split())
            prerequisites.append((x, y))

        target = int(input())

        result = build_time_dp(n, k, build_time, prerequisites, target)
        print(result)


if __name__ == '__main__':
    main()
